import java.io.Serializable;

class Student implements Serializable
{
	String name,school,stno;
	int grade;
	
	public Student(String name,String school,String stno,int grade)
	{
		this.name=name;
		this.school=school;
		this.stno=stno;
		this.grade=grade;
	}

}
服务端代码
import java.io.DataOutputStream;
import java.io.IOException;
import java.io.ObjectInputStream;
import java.net.ServerSocket;
import java.net.Socket;

public class Server
{

	public static void main(String[] args) throws IOException, ClassNotFoundException
	{
		ServerSocket serverSocket = new ServerSocket(10086);
		System.out.println("服务器已启动.");
		Socket socket = serverSocket.accept();
		ObjectInputStream ois = new ObjectInputStream(socket.getInputStream());
		DataOutputStream dos = new DataOutputStream(socket.getOutputStream());
	    Student stu =  (Student)(ois.readObject()); 
	    if (stu instanceof Student) 
	    {
	           System.out.println("name : " + stu.name);
	           System.out.println("school : " + stu.school);
	           System.out.println("stno : " + stu.stno);
	           System.out.println("grade : " + stu.grade);
	    }
	    dos.writeUTF(Integer.toString(stu.grade));
	    ois.close();
	    dos.close();
	    socket.close();
	    serverSocket.close();
	}
}

客户端代码
import java.io.DataInputStream;
import java.io.IOException;
import java.io.ObjectOutputStream;
import java.net.InetAddress;
import java.net.InetSocketAddress;
import java.net.Socket;
import java.net.UnknownHostException;

public class Client
{

	public static void main(String[] args) throws UnknownHostException, IOException
	{
		Student stu = new Student("ysy","sdust","201701060223",100);
		Socket socket = new Socket(InetAddress.getLocalHost(), 10086);
		System.out.println("已连接到服务器.");
		ObjectOutputStream oos = new ObjectOutputStream(socket.getOutputStream());
		DataInputStream dis = new DataInputStream(socket.getInputStream());
		
		oos.writeObject(stu);
		System.out.println("grade : " + dis.readUTF());
		
		oos.close();
		dis.close();
		socket.close();
	}

}
