#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <math.h>
#define OVERFLOW 9223372036854775807
long long fibonacci(int k)
{
   int i;
   long long x,y,z;
   x=1;
   y=1;
   z=1;
   for(i=2;i<=k;i++)
   {
       z = x+y;
       x=y;
       y=z;
   }
   return z;
}
int main()
{
    int n;
    long long fib;
    while(scanf("%d", &n) != EOF)
    {
        fib = fibonacci(n);
        if(fib == OVERFLOW)
            puts("overflow");
        else
            printf("%lld\n", fib);
    }
    return 0;
}

{
    int mm,dd,yyyy;
    int g,h;
    for(g=0;g<n-1;g++)
        for(h=0;h<n-1-g;h++)

            {
                yyyy=d[h].year;
                mm=d[h].month;
                dd=d[h].day;
