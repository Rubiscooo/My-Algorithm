#include <stdio.h>
#include <string.h>
#include <ctype.h>
#include <stdlib.h>
#include <math.h>
int f1()
{
    int a=123456;
    double b = 123.4567;
    char d = 'a';
    char c = 'A';  //给单个字符赋值用单引号
    char sstr[5] = "ABCD";
    char str[50] = "ABCD"; //给字符串数组赋值用双引号且要规定大小
    printf("%d %o %x %X %lf %e %g \n",a,a,a,a,b,b,b);
    // %d十进制 %o八进制 %x或%X十六进制  大小写对应
    //%lf输出double %e或%E 科学计数法 %g 按e,f较短的输出
    printf("%c %s %s\n",c,str,sstr);
    printf("%d %d\n",c,d);
    printf("%10d\n",a);  //默认右对齐
    printf("%-10d\n",a);   // -让输出结果左对齐
    printf("%15.8lf\n",b);  // m.n 同时控制宽度和精度
    printf("%-15.8lf\n",b);
    //字符是用ASCLL码存的用%d就能输出其对应的码 a = 65 A = 97
    putchar(c);
    puts(str);  //这两个输出函数不会自动换行
}

int f11()  //对结构体使用typedef的操作方式
{
    typedef struct
    {
        int year,month,date;
        char name[100];
    } DATE;
    int getDate(DATE *d)
    {
        scanf("%d %d %d %s",&d->year,&d->month,&d->date,d->name);
        puts(d->name); //输出name
    } //输入字符串不用+&
    int putDate(DATE d)
    {
        printf("%04d-%02d-%02d\n",d.year,d.month,d.date);
        puts(d.name);
    }
    int main()
    {
        DATE date;
        getDate(&date);
        putDate(date);
        return 0;
    }
}
