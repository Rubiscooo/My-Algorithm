int ackermann(int m,int n)
{
    if( m == 0)
        return n+1;
    if( m > 0 && n == 0)
        ackermann(m-1,1);
    if(m > 0 && n > 0)
        ackermann(m-1,ackermann(m,n-1));
}
int main()
{
  int m,n;
  int result;
  scanf("%d %d",&m,&n);
  result = ackermann(m,n);
  printf("A(%d, %d) = %d\n",m,n,result);
}

#define MAX_SIZE 200
 
int get_matrix(int a[][MAX_SIZE], int m, int n)
{
    int i, j;
    for (i = 1; i <= m; i++)
        for (j = 1; j <= n; j++)
            scanf("%d", &a[i][j]);
}
int count_sum(int arr[], int a[][MAX_SIZE], int m, int n)
{
    int i, j;
    for (i = 1; i <= m; i++)
    {
        int sum = 0;
        for (j = 1; j <= n; j++) sum += a[i][j];
        arr[i] = sum;
    }
}
 
int put_array(int arr[], int n)
{
    int i;
    for (i = 1; i < n; i++) printf("%d ", arr[i]);
    printf("%d\n", arr[n]);
}

第六题
#include<stdio.h>
#include<string.h>

typedef struct
{
    int ty;
    int point; // 点数
} Poker;

int get_card(Poker *pk)
{
    char s;
    s = getchar();
    if( s == 'A' )  pk->ty = 1;
    if( s == 'B' )  pk->ty = 2;
    if( s == 'C' )  pk->ty = 3;
    if( s == 'D' )  pk->ty = 4;
    getchar();
    scanf("%d",&pk->point);
    getchar();
}

int card_cmp(Poker p1, Poker p2)
{
    printf("%d %d\n",p1.ty,p1.point);
    printf("%d %d\n",p2.ty,p2.point);

    if(p1.point > p2.point)
        return 1;
    else if(p1.point < p2.point)
        return -1;
    else if(p1.point == p2.point)
    {
        if(p1.ty == p2.ty)
            return 0;
        if(p1.ty > p2.ty)
            return 1;
        if(p1.ty < p2.ty)
            return -1;
    }
}

int main()
{
    Poker tom, jack;
    get_card(&tom);
    get_card(&jack);

    int result = card_cmp(tom, jack);
    if(result > 0)
        puts("Tom wins.");
    if(result < 0)
        puts("Jack wins.");
    if(result == 0)
        puts("Tie.");
}

#include <stdio.h>
#include <string.h>
int main()
{
   int m,i,u=1,num,sum=0,result,sumzuo=0,sumyou=0,clas=0;
   int peo[27]={0};
   scanf("%d",&m);
    for(i=1; i<=m; i++)
    {
        scanf("%d",&peo[i]);
        sum += peo[i];
    }
   sumyou = peo[1];
   while(scanf("%d",&num) != EOF)
   {
      if( num == sum)
      {
            printf("Class %c : %d\n",'A' + m - 1 ,peo[m]);
            continue;
      }
       num = num % sum;
      if( num == 1)
        {
            printf("Class A : 1\n");
            continue;
        }

      while( num >= sumyou )
      {

          sumzuo += peo[u];
          sumyou += peo[u+1];
          u++;
          clas++;
      }
      result = num % sumzuo;
      if(result == 0)
        result = peo[u-1];
      printf("Class %c : %d\n",'A' + clas,result);
      u = 1;
      clas = 0;
      sumzuo = 0;
      sumyou = peo[1];
   }
}
