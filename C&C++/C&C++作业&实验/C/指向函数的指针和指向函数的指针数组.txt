int stcmp(void *a, void *b)
{
    char *x = (char *)a;
    char *y = (char *)b;
    if(strlen(x) > strlen(y)) return 1;
    if(strlen(x) < strlen(y)) return 0;
    return strcmp(x, y) > 0;
}

int icmp1(void *a, void *b)
{
    int *x = (int *)a;
    int *y = (int *)b;
    return *x > *y;
}

int icmp2(void *a, void *b)
{
    int *x = (int *)a;
    int *y = (int *)b;
    return *x < *y;
}

int dcmp(void *a, void *b)
{
    double *x = (double *)a;
    double *y = (double *)b;
    return *x < *y;
}

/*************************************************/
int swap(void *x, void *y, int size)
{
    char *t = (char *)malloc(size * sizeof(char));
    memcpy(t, x, size);
    memcpy(x, y, size);
    memcpy(y, t, size);
}

int sort(void *a, int size, int n, int (*cmp)(int *, int *))
{
    int i, j;
    for(i = 0; i < n - 1; i++)
        for(j = i + 1; j < n; j++)
            if(cmp(a + i * size, a + j * size))
                swap(a + i * size, a + j * size, size);
}
