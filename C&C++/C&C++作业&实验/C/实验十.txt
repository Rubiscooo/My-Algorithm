    ;

   {
       *(p + j) = *(s + i - 1 - j);
   }
   *(p + i)='\0';
   for(e=0;e<=i-1;e++)
   {
       *(s+e)=*(p+e);

第八题
#include <stdio.h>
#include <string.h>
#include <stdlib.h>

#define MAX_STR_LEN 11
#define MAX_SIZE 100
typedef struct
{
    int len;
    char buf[MAX_STR_LEN];
} string;
int swap(string *s1,string *s2)
{
    string tmp;
    strcpy(tmp.buf,s1->buf);
    strcpy(s1->buf,s2->buf);
    strcpy(s2->buf,tmp.buf);
    tmp.len = s2->len;
    s2->len = s1->len;
    s1->len = tmp.len;
    return 0;
}
int compare(string s1,string s2)
{
    if(s1.len > s2.len) return 1;
    if(s1.len < s2.len) return 0;
    if(strcmp(s1.buf,s2.buf) > 0)
        return 1;
    return 0;
}
int sort(string *ss,int n)
{
    int i,j;
    for(i=0; i<n-1; i++)
        for(j=i+1; j<n; j++)
            if(compare(ss[i],ss[j]))
                swap(&ss[i],&ss[j]);
    return 0;
}
int main()
{
    string strs[MAX_SIZE];
    int n,i;
    scanf("%d",&n);
    for(i=0; i<n; i++)
    {
        scanf("%s",strs[i].buf);
        strs[i].len = strlen(strs[i].buf);
    }
    sort(strs,n);
    for(i=0; i<n; i++)
        puts(strs[i].buf);
    return 0;
}
